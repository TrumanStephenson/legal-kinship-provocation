	id PRIMARY KEY AUTOINCREMENT,

	FOREIGN KEY (recognized_tribes_state) REFERENCES recognized_tribes(rec_location_state),

	FOREIGN KEY (lawyers_involved) REFERENCES lawyers(name)
	);

	FOREIGN KEY (lawyers_involved) REFERENCES lawyers(name)
	);
