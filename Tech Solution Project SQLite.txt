// Litigation Information
CREATE TABLE treaties (
	id PRIMARY KEY AUTOINCREMENT, 
	FOREIGN KEY (recognized_tribes_id) REFERENCES recognized_tribes(id), 
	FOREIGN KEY (recognized_tribes_state) REFERENCES recognized_tribes(rec_location_state),
	date TEXT,
	FOREIGN KEY (lawyers_involved) REFERENCES lawyers(name)
	);
	
CREATE TABLE landMarkCases (
	id PRIMARY KEY AUTOINCREMENT,
	FOREIGN KEY (recognized_tribes_id) REFERENCES recognized_tribes(id),
	FOREIGN KEY (recognized_tribes_state) REFERENCES recognized_tribes(rec_location_state),
	outcome TEXT,
	date TEXT,
	FOREIGN KEY (lawyers_involved) REFERENCES lawyers(name)
	);
	
CREATE TABLE lawyers (
	id PRIMARY KEY AUTOINCREMENT,
	lawyer_name TEXT,
	school TEXT,
	age INTEGER,
	active TEXT
	);
